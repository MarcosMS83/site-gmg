async function gerarPDF(){ const potencia = document.getElementById("potencia").value || "180"; const comprimento = parseInt( document.getElementById("comprimento").value || "3240" ); const largura = parseInt( document.getElementById("largura").value || "1130" ); const altura = parseInt( document.getElementById("alturaGmg").value || "1580" ); /* REGRAS AUTOMÁTICAS */ const comprimentoBase = comprimento + 2000; const larguraBase = largura + 2000; const existingPdfBytes = await fetch("Dimensional GMG 180KVA.pdf") .then(res => res.arrayBuffer()); const pdfDoc = await PDFLib.PDFDocument.load(existingPdfBytes); const pages = pdfDoc.getPages(); const firstPage = pages[0]; const white = PDFLib.rgb(1,1,1); const black = PDFLib.rgb(0,0,0); function apagar(x,y,w,h){ firstPage.drawRectangle({ x:x, y:y, width:w, height:h, color:white }); } /* APAGANDO TEXTOS ANTIGOS */ apagar(110,165,90,20); // 3130 apagar(290,80,90,20); // 5240 apagar(75,445,80,20); // 1130 apagar(470,445,80,20); // 5240 lateral apagar(630,330,80,20); // 1580 apagar(230,330,90,20); // 3240 frontal apagar(350,18,260,20); // titulo /* ESCREVENDO NOVOS VALORES */ /* 3240 -> comprimento */ firstPage.drawText(String(comprimento),{ x:235, y:333, size:12, color:black }); /* 5240 -> comprimento + 2000 */ firstPage.drawText(String(comprimentoBase),{ x:300, y:82, size:12, color:black }); firstPage.drawText(String(comprimentoBase),{ x:475, y:448, size:12, color:black }); /* 3130 -> largura + 2000 */ firstPage.drawText(String(larguraBase),{ x:120, y:168, size:12, color:black }); /* 1130 -> largura */ firstPage.drawText(String(largura),{ x:80, y:448, size:12, color:black }); /* 1580 -> altura */ firstPage.drawText(String(altura),{ x:635, y:333, size:12, color:black }); /* TITULO */ firstPage.drawText( `DIMENSIONAL COM BASE GMG ${potencia}KVA -`, { x:355, y:20, size:11, color:black } ); const pdfBytes = await pdfDoc.save(); download( pdfBytes, `Dimensional_GMG_${potencia}KVA.pdf`, "application/pdf" ); }
